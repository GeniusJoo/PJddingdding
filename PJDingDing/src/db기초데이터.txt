# 테이블 드랍(처음 생성이면 제거하고 실행)
drop Table school_information.library;
drop Table school_information.classroom;
drop Table school_information.student;
# 외래키 채킹해제
SET FOREIGN_KEY_CHECKS = 0;

#테이블 생성
CREATE TABLE school_information.student(
student_name  VARCHAR(10) COMMENT '학생이름',
student_id numeric(10) COMMENT '학생아이디',
student_password  VARCHAR(20) COMMENT '학생비번',
PRIMARY KEY(student_id)
)COMMENT = '학생정보';

CREATE TABLE school_information.classroom(
classroom_id numeric(10) COMMENT '강의실번호',
classroom_appointment numeric(10) DEFAULT '0' COMMENT '강의실예약자',
FOREIGN KEY (classroom_appointment) REFERENCES student (student_id),
PRIMARY KEY(classroom_id)
)COMMENT = '강의실정보';

CREATE TABLE school_information.library(
library_id numeric(10) COMMENT '도서관좌석',
library_user numeric(10) DEFAULT '0' COMMENT '좌석사용자',
Reservation_time time DEFAULT '00:00:00', 
FOREIGN KEY (library_user) REFERENCES student (student_id) ,
PRIMARY KEY(library_id)
)COMMENT = '도서관';
